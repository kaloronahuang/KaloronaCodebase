#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
const int MAX=55;
int p[MAX][MAX],k,tim=0,_max=0;
int m,n,total=0;
int main(){
    int x=1,y=1;
    int nx=0,ny=0;
    cin>>m>>n>>k;
    tim=k;
    for(int i=1;i<=m;i++){
        for(int j=1;j<=n;j++){
            cin>>p[i][j];		
            }
        }
    while(tim<k){
    int max=0,pi,pj;
    for(int i=1;i<=m;i++){
        for(int j=1;j<=n;j++){
            if(max<p[i][j]){
                max=p[i][j];
                pi=i;
                pj=j;
            }
        }
    }
    if(max==0) break;
    if(nx==0) ny=pj; 
    if(tim+pi+1+abs(pi-nx)+abs(pj-ny)<=k){
        tim+=1+abs(pi-nx)+abs(pj-ny);
        nx=pi;
        nx=pj;
        total+=p[pi][pj];
        p[pi][pj]=0;
        }
    else break;
    }
    printf("%d\n",total);
    return 0;
    }
